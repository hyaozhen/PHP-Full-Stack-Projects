create table grades ( pk_grade_ID INT not null auto_increment, student_id MEDIUMINT UNSIGNED not null, grade DECIMAL(5,2), school_code enum('L', 'B', 'A', 'F', 'E', 'T', 'I', 'W', 'S', 'U', 'M') not null, dept_id TINYINT UNSIGNED not null, course_code CHAR(5), foreign key(id) references students (id), foreign key(school_code,dept_id,course_code,name) references courses (school_code,dept_id,course_code,name)) engine = InnoDB default character set = utf8 collate = utf8_general_ci;
